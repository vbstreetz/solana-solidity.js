# Setup

This is a short guide into deploying and interacting with the standard ERC20 (opens new window) Solidity contract:

  1. Install docker (opens new window).
  2. Pull the required images:
docker pull solanalabs/solana:edge
docker pull hyperledgerlabs/solang:latest
  1. Start the test validator:
docker run --rm -it -p 8899:8899 -p 8900:8900 solanalabs/solana:edge > /dev/null
  1. Install the library:
npm install solana-solidity
  1. Compile the Solidity contract:
docker run --rm -it -v $PWD:/project --entrypoint /bin/bash hyperledgerlabs/solang -c "solang /project/contracts/ERC20.sol -o /project/build --target solana -v"

This outputs *.abi and bundle.so files in a build folder.

  1. Deploy and interact with the contract on Solana!
import { Program, newAccountWithLamports } from 'solana-solidity';
import CONTRACT_ABI from './build/ERC20.abi';
import PROGRAM_SO from './build/bundle.so'; // e.g. via webpack raw-loader
(async function () {
  const connection: Connection = new Connection(
    'http://localhost:8899',
    'confirmed'
  );
  const payerAccount = newAccountWithLamports();
  const program = await Program.deploy(connection, payerAccount, PROGRAM_SO);
  // deploy new contract
  const contractName = 'ERC20';
  const contractAbi = CONTRACT_ABI;
  const constructorArgs = [args...];
  const token1 = await program.deployContract(
    contractName,
    contractAbi,
    constructorArgs,
  );
  // call a function
  console.log(await token1.functions.symbol());
  // load an existing contract
  const token2 = await program.getContract(
    token.getStorageKeyPair(),
    CONTRACT_ABI
  );
  // subscribe to contract events
  const listenerId = token2.addEventListener('Approve', (owner: string, spender: string, value: string) => {
    console.log({
      owner,
      spender,
      value: : ethers.utils.formatEther(value),
    });
    token2.removeEventListener(listenerId);
  });
  // subscribe to "Program log:"s
  const logListenerId = program.addLogListener((msg: string) => {
    console.log(msg);
    program.removeLogListener(logListenerId);
  });
})();
